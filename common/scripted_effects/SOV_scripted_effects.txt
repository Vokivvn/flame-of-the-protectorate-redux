SOV_western_krai_lithuania_folds = {
	SOV = {
		if = {
			limit = {
				OR = {
					has_dlc = "Together for Victory"
					has_dlc = "Man the Guns"
				}
			}
			set_autonomy = {
				target = LIT
				autonomous_state = autonomy_rus_dutchy
			}
			else = { puppet = LIT }
		}
		add_to_faction = LIT
	}
	POL = { 
		country_event = { id = RUS.4 days = 1 }
	}
	LIT = {
		set_cosmetic_tag = LIT_RUS
		if = {
			limit = { has_war_with = POL }
			white_peace = POL
		}
		if = {
			limit = { has_war_with = BLR }
			white_peace = BLR
		}
		if = {
			limit = { has_war_with = UKR }
			white_peace = UKR
		}
		if = {
			limit = { has_war_with = UKN }
			white_peace = UKN
		}
		if = {
			limit = { 
				AND = { 
					SOV = { owns_state = 95 }
					SOV = { owns_state = 94 }
				}
			}
			LIT = { transfer_state = 95 transfer_state = 94 }
		}
	}	

}
