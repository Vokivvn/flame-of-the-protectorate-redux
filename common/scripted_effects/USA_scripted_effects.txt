USA_divide_house_seats_on_parties = {
	#log = "state_max_representative = [?state_max_representative]"
	set_variable = { USA_PROGESSIVE_HOUSE_SEATS = US_progressive_popularity_election }
	set_variable = { USA_DEMOCRAT_HOUSE_SEATS = US_democrat_popularity_election }
	set_variable = { USA_WHIG_HOUSE_SEATS = US_whig_popularity_election }
	set_variable = { USA_REPUBLICAN_HOUSE_SEATS = US_socialist_popularity_election }

	multiply_variable = { USA_PROGESSIVE_HOUSE_SEATS = state_max_representative }
	multiply_variable = { USA_DEMOCRAT_HOUSE_SEATS = state_max_representative }
	multiply_variable = { USA_WHIG_HOUSE_SEATS = state_max_representative }
	multiply_variable = { USA_REPUBLICAN_HOUSE_SEATS = state_max_representative }

	round_variable = USA_PROGESSIVE_HOUSE_SEATS
	round_variable = USA_DEMOCRAT_HOUSE_SEATS
	round_variable = USA_WHIG_HOUSE_SEATS
	round_variable = USA_REPUBLICAN_HOUSE_SEATS

	set_temp_variable = { total_representatives = USA_PROGESSIVE_HOUSE_SEATS }
	add_to_temp_variable = { total_representatives = USA_DEMOCRAT_HOUSE_SEATS }
	add_to_temp_variable = { total_representatives = USA_WHIG_HOUSE_SEATS }
	add_to_temp_variable = { total_representatives = USA_REPUBLICAN_HOUSE_SEATS }

	if = {
		limit = { NOT = { check_variable = { total_representatives = state_max_representative}}}
		if = { #too many are delegated
			limit = { check_variable = { total_representatives > state_max_representative }}
			if = {
				limit = { check_variable = { largest_party = 1 }}
				add_to_variable = { USA_PROGESSIVE_HOUSE_SEATS = -1}
			}
			if = {
				limit = { check_variable = { largest_party = 2 }}
				add_to_variable = { USA_DEMOCRAT_HOUSE_SEATS = -1}
			}
			if = {
				limit = { check_variable = { largest_party = 3 }}
				add_to_variable = { USA_WHIG_HOUSE_SEATS = -1}
			}
			if = {
				limit = { check_variable = { largest_party = 4 }}
				add_to_variable = { USA_REPUBLICAN_HOUSE_SEATS = -1}
			}
		}
		if = { #too few are delegated
			limit = { check_variable = { total_representatives < state_max_representative }}
			if = {
				limit = { check_variable = { smallest_party = 1 }}
				add_to_variable = { USA_PROGESSIVE_HOUSE_SEATS = 1}
			}
			if = {
				limit = { check_variable = { smallest_party = 2 }}
				add_to_variable = { USA_DEMOCRAT_HOUSE_SEATS = 1}
			}
			if = {
				limit = { check_variable = { smallest_party = 3 }}

			}
			if = {
				limit = { check_variable = { smallest_party = 4 }}
				add_to_variable = { USA_REPUBLICAN_HOUSE_SEATS = 1}
			}
		}	
	}
